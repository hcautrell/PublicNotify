    FOR I in 1..APEX_APPLICATION.G_F01.COUNT LOOP

                  SET INACTIVE_FLAG = 'Y',
		         USER_LAST_UPDT = :APP_USER,

            WHERE REQUIREMENT_ID = APEX_APPLICATION.G_F02(I2);
        END LOOP;
    END LOOP;
END;

    APEX_ITEM.TEXT(5,SAL) SAL, 
    APEX_ITEM.TEXT(6,COMM) COMM,
    DEPTNO
FROM EMP
